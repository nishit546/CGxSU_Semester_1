

### React Component (Products from Fake Store API)

You can put this in a file like `FakeStoreProducts.jsx` and render it in your app.

```jsx
import React, { useState } from "react";

function FakeStoreProducts() {
  const [products, setProducts] = useState([]);
  const [error, setError] = useState("");

  const handleLoadProducts = () => {
    setError("");
    fetch("https://fakestoreapi.com/products")
      .then((response) => response.json())
      .then((data) => {
        console.log("All products:", data);
        setProducts(data);
      })
      .catch((err) => {
        console.log("Error:", err);
        setError("Error loading products.");
      });
  };

  return (
    <div>
      <h1>Fake Store Products</h1>

      <button onClick={handleLoadProducts}>Load Products</button>

      {error && <p>{error}</p>}

      <div>
        {products.map((p) => (
          <div key={p.id}>
            <h3>{p.title}</h3>
            <img src={p.image} alt={p.title} width="100" />
            <p>Price: ${p.price}</p>
            <p>Category: {p.category}</p>
            <p>{p.description}</p>
            <hr />
          </div>
        ))}
      </div>
    </div>
  );
}

export default FakeStoreProducts;
```

---

This will give you almost the **same output structure** as your plain HTML+JS version, but in a clean React component.
